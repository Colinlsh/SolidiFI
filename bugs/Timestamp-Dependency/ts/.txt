Usage: <main class> [options]
  Options:
    -co, --compilationoutput
       compilation output of a project
    --decompoutputfile
       output file for the decompiled code
    --descriptions
       add descriptions to the JSON output
       Default: false
    -fh, --filehex
       contract runtime code to parse as a hex-encoded file
    -fs, --filesol
       smart contract as a Solidity file
    -h, -?, --help
       usage
       Default: false
    --json
       provide JSON output to console
       Default: false
    --livestatusfile
       status output file to track analysis progress
       Default: /tmp.txt
    -o, --output
       json output file
    -p, --patterns
       csv list of patterns to be analyzed
    -q, --quiet
       suppress most output
       Default: false
    --solc-path
       specify the path of the solc binary
       Default: solc
    -v, --verbose
       provide verbose output
       Default: false
